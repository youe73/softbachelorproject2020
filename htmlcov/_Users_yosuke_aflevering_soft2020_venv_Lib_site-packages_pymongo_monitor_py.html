56class Monitor(MonitorBase): 

57 def __init__( 

58 self, 

59 server_description, 

60 topology, 

61 pool, 

62 topology_settings): 

63 """Class to monitor a MongoDB server on a background thread. 

64 

65 Pass an initial ServerDescription, a Topology, a Pool, and 

66 TopologySettings. 

67 

68 The Topology is weakly referenced. The Pool must be exclusive to this 

69 Monitor. 

70 """ 

71 self._server_description = server_description 

72 self._pool = pool 

73 self._settings = topology_settings 

74 self._avg_round_trip_time = MovingAverage() 

75 self._listeners = self._settings._pool_options.event_listeners 

76 pub = self._listeners is not None 

77 self._publish = pub and self._listeners.enabled_for_server_heartbeat 

78 

79 # We strongly reference the executor and it weakly references us via 

80 # this closure. When the monitor is freed, stop the executor soon. 

81 def target(): 

82 monitor = self_ref() 

83 if monitor is None: 

84 return False # Stop the executor. 

85 Monitor._run(monitor) 

86 return True 

87 

88 executor = periodic_executor.PeriodicExecutor( 

89 interval=self._settings.heartbeat_frequency, 

90 min_interval=common.MIN_HEARTBEAT_INTERVAL, 

91 target=target, 

92 name="pymongo_server_monitor_thread") 

93 

94 self._executor = executor 

95 

96 # Avoid cycles. When self or topology is freed, stop executor soon. 

97 self_ref = weakref.ref(self, executor.close) 

98 self._topology = weakref.proxy(topology, executor.close) 

99 

100 def close(self): 

101 super(Monitor, self).close() 

102 

103 # Increment the pool_id and maybe close the socket. If the executor 

104 # thread has the socket checked out, it will be closed when checked in. 

105 self._pool.reset()

107 def _run(self): 

108 try: 

109 self._server_description = self._check_with_retry() 

110 self._topology.on_change(self._server_description) 

111 except ReferenceError: 

112 # Topology was garbage-collected. 

113 self.close()

115 def _check_with_retry(self): 

116 """Call ismaster once or twice. Reset server's pool on error. 

117 

118 Returns a ServerDescription. 

119 """ 

120 # According to the spec, if an ismaster call fails we reset the 

121 # server's pool. If a server was once connected, change its type 

122 # to Unknown only after retrying once. 

123 address = self._server_description.address 

124 retry = True 

125 if self._server_description.server_type == SERVER_TYPE.Unknown: 

126 retry = False 

127 

128 start = _time() 

129 try: 

130 return self._check_once() 

131 except ReferenceError: 

132 raise 

133 except Exception as error: 

134 error_time = _time() - start 

135 if self._publish: 

136 self._listeners.publish_server_heartbeat_failed( 

137 address, error_time, error) 

138 self._topology.reset_pool(address) 

139 default = ServerDescription(address, error=error) 

140 if not retry: 

141 self._avg_round_trip_time.reset() 

142 # Server type defaults to Unknown. 

143 return default 

144 

145 # Try a second and final time. If it fails return original error. 

146 # Always send metadata: this is a new connection. 

147 start = _time() 

148 try: 

149 return self._check_once() 

150 except ReferenceError: 

151 raise 

152 except Exception as error: 

153 error_time = _time() - start 

154 if self._publish: 

155 self._listeners.publish_server_heartbeat_failed( 

156 address, error_time, error) 

157 self._avg_round_trip_time.reset() 

158 return default 

159 

160 def _check_once(self): 

161 """A single attempt to call ismaster. 

162 

163 Returns a ServerDescription, or raises an exception. 

164 """ 

165 address = self._server_description.address 

166 if self._publish: 

167 self._listeners.publish_server_heartbeat_started(address) 

168 with self._pool.get_socket({}) as sock_info: 

169 response, round_trip_time = self._check_with_socket(sock_info) 

170 self._avg_round_trip_time.add_sample(round_trip_time) 

171 sd = ServerDescription( 

172 address=address, 

173 ismaster=response, 

174 round_trip_time=self._avg_round_trip_time.get()) 

175 if self._publish: 

176 self._listeners.publish_server_heartbeat_succeeded( 

177 address, round_trip_time, response) 

178 

179 return sd 

180 

181 def _check_with_socket(self, sock_info): 

182 """Return (IsMaster, round_trip_time). 

183 

184 Can raise ConnectionFailure or OperationFailure. 

185 """ 

186 start = _time() 

187 try: 

188 return (sock_info.ismaster(self._pool.opts.metadata, 

189 self._topology.max_cluster_time()), 

190 _time() - start) 

191 except OperationFailure as exc: 

192 # Update max cluster time even when isMaster fails. 

193 self._topology.receive_cluster_time( 

194 exc.details.get('$clusterTime')) 

195 raise

231 def _run(self): 

232 seedlist = self._get_seedlist() 

233 if seedlist: 

234 self._seedlist = seedlist 

235 try: 

236 self._topology.on_srv_update(self._seedlist) 

237 except ReferenceError: 

238 # Topology was garbage-collected. 

239 self.close()

241 def _get_seedlist(self): 

242 """Poll SRV records for a seedlist. 

243 

244 Returns a list of ServerDescriptions. 

245 """ 

246 try: 

247 seedlist, ttl = _SrvResolver(self._fqdn).get_hosts_and_min_ttl() 

248 if len(seedlist) == 0: 

249 # As per the spec: this should be treated as a failure. 

250 raise Exception 

251 except Exception: 

252 # As per the spec, upon encountering an error: 

253 # - An error must not be raised 

254 # - SRV records must be rescanned every heartbeatFrequencyMS 

255 # - Topology must be left unchanged 

256 self.request_check() 

257 return None 

258 else: 

259 self._executor.update_interval( 

260 max(ttl, common.MIN_SRV_RESCAN_INTERVAL)) 

261 return seedlist
